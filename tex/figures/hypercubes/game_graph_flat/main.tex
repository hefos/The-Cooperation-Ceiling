\begin{tikzpicture}[scale=2,
    vertex/.style={circle, draw, minimum size=6mm, inner sep=0pt, font=\small},
    edge/.style={->, thick},
    loop/.style={->, looseness=4, min distance=3.5mm}
]

% --- Vertices by Hamming weight ---

% Weight 0
\node[vertex] (000) at (0,1.5) {$000$};

% Weight 1
\node[vertex] (100) at (-1,1) {$100$};
\node[vertex] (010) at (0,1) {$010$};
\node[vertex] (001) at (1,1) {$001$};

% Weight 2
\node[vertex] (110) at (-1,0.5) {$110$};
\node[vertex] (101) at (0,0.5) {$101$};
\node[vertex] (011) at (1,0.5) {$011$};

% Weight 3
\node[vertex] (111) at (0,0) {$111$};

% --- Bidirectional edge function ---
\newcommand{\bidedge}[2]{
    \draw[edge, bend left=10] (#1) to (#2);
    \draw[edge, bend left=10] (#2) to (#1);
}

% --- Edges between vertices differing by 1 bit ---
\bidedge{000}{100}
\bidedge{000}{010}
\bidedge{000}{001}

\bidedge{100}{110}
\bidedge{100}{101}
\bidedge{010}{110}
\bidedge{010}{011}
\bidedge{001}{101}
\bidedge{001}{011}

\bidedge{110}{111}
\bidedge{101}{111}
\bidedge{011}{111}

% --- Self-loops placed “outside” the main graph to avoid crossing edges ---

% Weight 0
\draw[loop, out=135, in=45] (000) to (000);

% Weight 1
\draw[loop, out=180, in=120] (100) to (100); % left vertex → left side
\draw[loop, out=90, in=0]   (010) to (010); % middle vertex → top side
\draw[loop, out=60, in=0]   (001) to (001); % right vertex → right-top

% Weight 2
\draw[loop, out=180, in=150] (110) to (110); % left → left-top
\draw[loop, out=90, in=30]   (101) to (101); % middle → top
\draw[loop, out=30, in=330] (011) to (011); % right → right-top

% Weight 3
\draw[loop, out=225, in=135] (111) to (111); % bottom vertex → bottom-left side

\end{tikzpicture}
