\begin{tikzpicture}[scale=2]

\coordinate (C) at (0,0);
\draw[thick] (C) ellipse (00.8 and 0.35);

\tikzset{
    token/.style={thick},
}

% --- Person with coin ---
\newcommand{\person}{
    \draw (0,0.11) circle (0.04);
    \draw (0,0.07) -- (0,-0.05);
    \draw (-0.06,0.03) -- (0.06,0.03);
    \coordinate (hand) at (0.08,0.03);
    \draw[token] (hand) circle (0.035);
    \draw (0,-0.05) -- (-0.04,-0.11);
    \draw (0,-0.05) -- (0.04,-0.11);
}

% --- Place 6 people (slightly closer too) ---
\foreach \angle in {0,60,120,180,240,300}
{
    \path ({1.15*cos(\angle)},{0.7*sin(\angle)}) coordinate (P);

    \begin{scope}[shift={(P)}]
        \person

        % Three contributors
        \ifnum\angle=0
            \draw[->, thick] (hand) -- ($(hand)!0.75!(C)$);
        \else\ifnum\angle=120
            \draw[->, thick] (hand) -- ($(hand)!0.75!(C)$);
        \else\ifnum\angle=240
            \draw[->, thick] (hand) -- ($(hand)!0.75!(C)$);
        \fi\fi\fi

    \end{scope}
}

\end{tikzpicture}
